  <script>
    function refresh() {
      fetch("/api/run")
        .then(r => r.json())
        .then(data => {
          const tbody = document.querySelector("tbody");
          tbody.innerHTML = "";
          if (data.length === 0) {
            tbody.innerHTML = "<tr><td colspan='4'>No stocks matched today.</td></tr>";
          } else {
            document.querySelector("h2").innerText = "Matching stocks: " + data.length;
            data.forEach(r => {
              tbody.innerHTML += `
                <tr>
                  <td>${r.ticker}</td>
                  <td>${r.close}</td>
                  <td>${r.ema20}</td>
                  <td>${r.ema50}</td>
                </tr>
              `;
            });
          }
        });
    }
